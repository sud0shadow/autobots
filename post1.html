Hate repetitive work? These Python scripts save time.

1. Batch Rename Files


import os

folder = "path/to/folder"
for i, filename in enumerate(os.listdir(folder)):
    os.rename(f"{folder}/{filename}", f"{folder}/file_{i}.txt")
print("Done renaming!")

2. Email Alert

        
import smtplib
from email.mime.text import MIMEText    

msg = MIMEText("Task complete!")
msg["Subject"] = "Alert"
msg["From"] = "you@gmail.com"
msg["To"] = "you@gmail.com"

with smtplib.SMTP("smtp.gmail.com", 587) as server:
    server.starttls()
    server.login("you@gmail.com", "your_password")
    server.send_message(msg)

print("Email sent!")

3. Simple Web Scraper


import requests
from bs4 import BeautifulSoup

url = "https://example.com"
response = requests.get(url)
soup = BeautifulSoup(response.text, "html.parser")

titles = soup.find_all("h1")
for title in titles:
    print(title.text)
